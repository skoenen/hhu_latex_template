\usepackage{listings}
\lstset{numbers=left}
\lstset{%
	basicstyle=\small,
	keywordstyle=\color{DodgerBlue2},
	identifierstyle=,
	commentstyle=\color{Green4}\itshape,
	stringstyle={\color{violet}\ttfamily},
	showstringspaces=true,
	breaklines=true
}


\newcommand{\inifile}[1]{%
	\lstinputlisting[
		caption={\detokinze{#5}},
		language=command.com,
		tabsize=4
	]{#1}
	\label{lst:ini_#1}
}

\newcommand{\inisnip}[5]{%
	\lstinputlisting[%
		caption={\detokinze{#5}},
		language=command.com,
		tabsize=4,
		firstline=#2,
		firstnumber=#2,
		lastline=#3
	]{#1}
	\label{lst:ini_#4}
}
