\usepackage{listings}
\lstset{numbers=left}
\lstset{%
	basicstyle=\small,
	keywordstyle=\color{DodgerBlue2},
	identifierstyle=,
	commentstyle=\color{Green4}\itshape,
	stringstyle={\color{violet}\ttfamily},
	showstringspaces=true,
	breaklines=true
}


\newcommand{\inifile}[1]{%
	\lstinputlisting[
		caption={\normalsize{#1}},
		label={lst:ini_#1},
		language=command.com,
		tabsize=4
	]{#1}
}

\newcommand{\inisnip}[4]{%
	\lstinputlisting[%
		caption={\normalsize{#4}},
		label={lst:ini_#3},
		language=command.com,
		tabsize=4,
		linerange={#2},
	]{#1}
}

\newcommand{\inicode}[3]{%
	\begin{lstlisting}[%
		caption={\normalsize{#3}},
		label={lst:ini_#2},
		language=command.com,
		tabsize=4]
#1
	\end{lstlisting}
}
